TOP 10. NOIP2012 疫情控制

花了一上午敲完又花了一节课卡进1s其实是因为心态崩了干脆开了O2优化

这道题的想法是

(暴力)

枚举所花费的时间,然后让军队暴力地向上跳

发现这两步都能优化

如果TTT秒可以部署完所有军队,T+1T+1T+1显然也行,只是浪费一些

据此可以二分答案.

然后跳跃也可以像找LCA那样倍增预处理

然后还需要一步贪心:

对于一个点,如果军队还能走的步数多,那么优先考虑用这些军队去建立检查站.

TOP 8 NOIP2012 开车旅行

70%直接模拟

100% 维护每个人走2的若干次方步能到哪里.

注意内存.

	#include<algorithm>
#include<cctype>
#include<cmath>

#include<cstdlib>
#include<cstring>
#include<iostream>
#include<ctime>
#include<set>
#include<queue>
#include<stack>
#include<vector>
#define size 300010
#define inf 10000000000000
#define gc getchar()
#define db double
#define ll long long
#define rep(i,s,n) for (register ll i=s;i<=n;i++)
#define drep(i,n,s) for (register ll i=n;i>=s;i--)
#define il inline
using namespace std;

il ll r()
{
    char c; ll x,f=1;
    for (c=gc;!isdigit(c);c=gc) if (c=='-') f=-1; x=c-'0';
    for (c=gc;isdigit(c);c=gc) x=(x<<1)+(x<<3)+c-'0';
    return f*x;
}

struct city
{
    ll id; ll alt;
    friend bool operator<(city a,city b)
    {
        return a.alt<b.alt;
    }
};

ll n;
ll a[size],went[size][25][5],ago[size][25][3],bgo[size][25][3],X0;
multiset<city> M;

void calc(ll S,ll &disa,ll &disb,ll x0)
{
    ll now=S;
    for (ll k=18;k>=0;k--)
        if (went[now][k][0]&&disa+disb+ago[now][k][0]+bgo[now][k][0]<=x0)
        {
            disa+=ago[now][k][0];
            disb+=bgo[now][k][0];
            now=went[now][k][0];
        }
}

void init()
{
    for(ll i=n;i;i--)
    {
        ll goa,gob;
       	city aa;
        aa.id=i,aa.alt=a[i];
        M.insert(aa);
        multiset<city>::iterator it=M.lower_bound(aa);
        ll pre,nxt;
        ll preh,nxth;
        it++; nxt=(*it).id; nxth=(*it).alt; 
        it--; it--; pre=(*it).id; preh=(*it).alt; 
        it++;
        if(abs(nxth-a[i])>=abs(preh-a[i])){
            gob=pre;
            it--;it--;
            if(abs(nxth-a[i])>=abs((*it).alt-a[i])) goa=(*it).id;
            else goa=nxt;
        }
        else{
            gob=nxt;
            it++;it++;
            if(abs(preh-a[i])>abs((*it).alt-a[i])) goa=(*it).id;
            else goa=pre;
        }
        went[i][0][0]=goa;
        went[i][0][1]=gob;
        ago[i][0][0]=abs(a[i]-a[goa]);
        ago[i][1][0]=ago[i][0][0];
        bgo[i][0][1]=abs(a[i]-a[gob]);
        bgo[i][1][1]=bgo[i][0][1];
        
    }
    rep(i,1,n)
    {
        went[i][1][0]=went[went[i][0][0]][0][1];
        ago[i][1][0]=ago[i][0][0];
        bgo[i][1][0]=abs(a[went[i][1][0]]-a[went[i][0][0]]);
        went[i][1][1]=went[went[i][0][1]][0][0];
        bgo[i][1][1]=bgo[i][0][1];
        ago[i][1][1]=abs(a[went[i][1][1]]-a[went[i][0][1]]);
    }
}

void bz()
{
    rep(k,2,18)
        rep(i,1,n)
        {
            went[i][k][1]=went[went[i][k-1][1]][k-1][1];
            bgo[i][k][1]=bgo[went[i][k-1][1]][k-1][1]+bgo[i][k-1][1];
            ago[i][k][1]=ago[went[i][k-1][1]][k-1][1]+ago[i][k-1][1];
            went[i][k][0]=went[went[i][k-1][0]][k-1][0];
            bgo[i][k][0]=bgo[went[i][k-1][0]][k-1][0]+bgo[i][k-1][0];
            ago[i][k][0]=ago[went[i][k-1][0]][k-1][0]+ago[i][k-1][0];
        }
}


int main()
{
    
    n=r(); rep(i,1,n) a[i]=r(); a[0]=inf; a[n+1]=-inf;
    M.insert((city){0,inf}); M.insert((city){0,inf});
    M.insert((city){n+1,-inf}); M.insert((city){n+1,-inf});
    init(); bz();
    X0=r(); db mn=1e16;ll Id,T;
    rep(i,1,n)
    {
        ll la=0,lb=0;
        calc(i,la,lb,X0);
        if (!lb)
        {
            if (mn>1e15) mn=1e15,Id=i;
                else if (mn==1e15&&a[Id]<a[i]) Id=i;
        }
        else 
        {
            db now=(db)la/(db)lb;
            if (now<mn) mn=now,Id=i;
            else if (now==mn&&a[Id]<a[i]) Id=i;
        }
    }
    printf("%lld\n",Id);
    T=r();
    while (T--)
    {
        ll a=r(),b=r(); ll la=0,lb=0;
        calc(a,la,lb,b);
        printf("%lld %lld\n",la,lb);
    }
    return 0;
}
